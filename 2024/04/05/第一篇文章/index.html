链表

自己写的链表demo

  1.  struct ListNode {
      int val;
     ListNode *next;
     ListNode() : val(0), next(nullptr) {}
      ListNode(int x) : val(x), next(nullptr) {}
      ListNode(int x, ListNode *next) : val(x), next(next) {}
  };
 
  int main() {
      ListNode* head = new ListNode(4);
     ListNode* p = head;
 
      for (int i = 0; i < 5; i++) {
          ListNode* q = new ListNode(i + 1);
          p->next = q;
          p = p->next;
      }
       p = head;
      while (p) {
          cout << p->val << "-->";
          p = p->next;
      }
      return 0;
  }
